form.elements[0]

4，避免多次提交表单

EventUtil.addHandler(form,"submit",function(event){
	event = EventUtil.getTarget(event);
	var target = EventUtil.getTarget(event);
	var btn = target.elements["submit-btn"];
	btn.disabled = true;
});
